extends Icons.BlockIcon;

  outer PDE.World.worldModel worldModel1;
  parameter Integer n=worldModel1.n;

equation 
for j in 1:n+1 loop
if u2[j] > 0 then
  for i in 1:m loop
    y[i, j] = u[j]*u1[i, j];
    y1[i, j] = 0.0;
  end for;
else
  for i in 1:m loop
    y1[i, j] = u[j]*u1[i, j];
    y[i, j] = 0.0;
  end for;
end if;
end for;
