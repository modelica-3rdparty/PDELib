  Modelica.Blocks.Interfaces.RealInput u1[worldModel1.n + worldModel1.

    gcl + worldModel1.gcr];
equation 
for j in 1:n+gcl+gcr-1 loop
  y[1, 1, j] = 1.0;
  y[2, 1, j] = u[j] - u1[j];
  y[3, 1, j] = u2[j] - u[j]*u1[j];

  y[1, 2, j] = 1.0;
  y[2, 2, j] = u[j];
  y[3, 2, j] = 0.5*((u[j])^2);

  y[1, 3, j] = 1.0;
  y[2, 3, j] = u[j] + u1[j];
  y[3, 3, j] = u2[j] + u[j]*u1[j];
end for;
