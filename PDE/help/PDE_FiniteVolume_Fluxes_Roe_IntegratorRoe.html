    "|Boundary Conditions| Boundary condition at the right (0: no; 1: yes)";

    "|Boundary Conditions| Number of ghost cells at the left";

equation 
  y = q;

for i in 1:m loop
  for j in 1:gcl loop
     q[i, j]= u2[i, j];
  end for;
end for;

for i in 1:m loop
  if bcl == 1 then
     q[i, gcl+1] = q[i, gcl];
  end if;
end for;

for i in 1:m loop
  for j in 1:gcr loop
     q[i, gcl+n+j] = u3[i, j];
  end for;
end for;

for i in 1:m loop
  if bcr == 1 then
     q[i, gcl+n] = q[i, gcl+n+1];
  end if;
end for;

for j in vb:ve loop
  for i in 1:m loop
    der(q[i, j]) = -(1/delta_x)*(u[i, j-gcl] + u4[i, j-gcl+1]);
  end for;
end for;
